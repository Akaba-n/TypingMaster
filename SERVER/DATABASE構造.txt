//*---------- ユーザー情報格納DB ----------*//

CREATE DATABASE UserDB default character set utf8mb4;
//*----- ユーザー情報テーブル -----*//
// テーブル作成
CREATE TABLE User(
userId VARCHAR(10) NOT NULL PRIMARY KEY,
userName VARCHAR(10) NOT NULL,
mailAddress VARCHAR(255) NOT NULL UNIQUE,
lastLogin DateTime
)
default character set utf8mb4;
