//*---------- ユーザー情報格納DB ----------*//
// データベース作成
CREATE DATABASE TypingDB default character set utf8mb4;

//*----- ユーザー情報テーブル -----*//
// テーブル作成
// パスワードはハッシュ化する
// INTEGER(符号付き整数型)では外部キー参照出来ないのでUNSIGNEDINTEGER(符号なし整数型)を使う
CREATE TABLE User(
userId INTEGER(8) UNSIGNED ZEROFILL AUTO_INCREMENT NOT NULL PRIMARY KEY,
userName VARCHAR(10) NOT NULL,
email VARCHAR(255) NOT NULL UNIQUE,
pass VARCHAR(255) NOT NULL
)
default character set utf8mb4;


//*----- ログインテーブル -----*//
// ここでIP(接続元)の管理も行えば同時ログインの管理も出来る...?
CREATE TABLE Login(
userId INTEGER(8) UNSIGNED ZEROFILL,
logInTime DateTime,
CONSTRAINT fk_login_userId
	FOREIGN KEY(userId)
	REFERENCES User(userId)
	ON DELETE CASCADE ON UPDATE CASCADE
)
default character set utf8mb4;


//*----- 記録格納テーブル -----*//
// テーブル作成
CREATE TABLE Record(
recId INTEGER(255) AUTO_INCREMENT PRIMARY KEY,
userId INTEGER(8) UNSIGNED ZEROFILL,
recTime DateTime,
CONSTRAINT fk_rec_userId
	FOREIGN KEY(userId)
	REFERENCES User(userId)
	ON DELETE CASCADE ON UPDATE CASCADE
)
default character set utf8mb4;
